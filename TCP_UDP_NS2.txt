# Create simulator
set ns [new Simulator]

# Define colors
$ns color 1 Blue
$ns color 2 Red

# Open NAM trace file
set nf [open out.nam w]
$ns namtrace-all $nf

# Finish procedure

proc finish {} {
global ns nf
$ns flush-trace
close $nf
exec nam out.nam &amp;
exit 0
}

# Create nodes
set n0 [$ns node]
set n1 [$ns node]
set n2 [$ns node]
set n3 [$ns node]

# Create links
$ns duplex-link $n0 $n2 2Mb 10ms DropTail
$ns duplex-link $n1 $n2 2Mb 10ms DropTail
$ns duplex-link $n2 $n3 1.7Mb 20ms DropTail

# Queue limit
$ns queue-limit $n2 $n3 10

# Link orientation (for NAM)
$ns duplex-link-op $n0 $n2 orient right-down
$ns duplex-link-op $n1 $n2 orient right-up
$ns duplex-link-op $n2 $n3 orient right
$ns duplex-link-op $n2 $n3 queuePos 0.5

# TCP setup
set tcp [new Agent/TCP]
$tcp set class_ 2
$tcp set fid_ 1

$ns attach-agent $n0 $tcp

set sink [new Agent/TCPSink]
$ns attach-agent $n3 $sink

$ns connect $tcp $sink

# FTP over TCP
set ftp [new Application/FTP]
$ftp attach-agent $tcp
$ftp set type_ FTP

# UDP setup
set udp [new Agent/UDP]
$udp set fid_ 2
$ns attach-agent $n1 $udp

set null [new Agent/Null]
$ns attach-agent $n3 $null

$ns connect $udp $null

# CBR over UDP

set cbr [new Application/Traffic/CBR]
$cbr attach-agent $udp
$cbr set type_ CBR
$cbr set packet_size_ 1000
$cbr set rate_ 1mb
$cbr set random_ false

# Scheduling events
$ns at 0.1 &quot;$cbr start&quot;
$ns at 1.0 &quot;$ftp start&quot;
$ns at 4.0 &quot;$ftp stop&quot;
$ns at 4.5 &quot;$cbr stop&quot;

# Detach agents
$ns at 4.5 &quot;$ns detach-agent $n0 $tcp&quot;
$ns at 4.5 &quot;$ns detach-agent $n3 $sink&quot;

# End simulation
$ns at 5.0 &quot;finish&quot;

# Print values
puts &quot;CBR packet size = [$cbr set packet_size_]&quot;
puts &quot;CBR interval = [$cbr set interval_]&quot;

# Run simulation
$ns run
